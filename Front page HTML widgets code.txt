<script>
document.addEventListener('DOMContentLoaded', function() {
    // Find the form and buttons
    const form = document.getElementById('signup-form');
    const coreButton = document.getElementById('core-subscribe-button');
    const enhancedButton = document.getElementById('enhanced-subscribe-button');
    
    if (!form || (!coreButton && !enhancedButton)) {
        console.log("Form or buttons not found");
        return;
    }
    
    // Check if all required fields are filled
    function isFormComplete() {
        const requiredFields = form.querySelectorAll('input[required]');
        for (let field of requiredFields) {
            if (!field.value.trim()) {
                return false;
            }
        }
        return true;
    }
    
    // Add click handlers to both buttons
    function addButtonHandler(button, productId) {
        if (!button) return;
        
        // Store original href
        const originalHref = button.href;
        
        // Add click event handler
        button.addEventListener('click', function(e) {
            // If form is not complete, prevent navigation
            if (!isFormComplete()) {
                e.preventDefault();
                alert("Please complete all required fields before subscribing.");
                return false;
            }
            
            // Store form data
            const formData = {};
            const inputs = form.querySelectorAll('input');
            inputs.forEach(input => {
                formData[input.id] = input.value;
            });
            
            // Save to session storage
            sessionStorage.setItem('cd_registration_data', JSON.stringify(formData));
            
            // Let the navigation proceed
            return true;
        });
    }
    
    // Add handlers to buttons
    addButtonHandler(coreButton, 846);
    addButtonHandler(enhancedButton, 847);
    
    console.log("Button handlers added successfully");
});
</script>
